% allows elements in matrices and vertically stretching an array.

% argument 1 (optional) is the array stretch amount (if needed, 1.5 usually suffices)
% argument 2 (required) is the alignment of elements in the matrix
% i would have made both optional but i could not figure out how to get it to work
% https://texblog.net/latex-archive/uncategorized/matrix-align-left-right/
% https://tex.stackexchange.com/a/14072
% https://stackoverflow.com/a/1812224
\makeatletter
\renewcommand*\env@matrix[2][\arraystretch]{
    \edef\arraystretch{#1}
    \hskip -\arraycolsep
    \let\@ifnextchar\new@ifnextchar
    \array{*\c@MaxMatrixCols #2}}
\makeatother
